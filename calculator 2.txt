pragma solidity >=0.8.2 <0.9.0;

contract Storage {
    function addition(uint256 number1,uint256 number2) public pure  returns (uint256){
        return number1+number2;
    }

    function subtraction(uint256 number1,uint256 number2) public pure  returns (uint256){
        return number1-number2;
    }
    function multiplication(uint256 number1,uint256 number2) public pure  returns (uint256){
        return number1*number2;
    }

    function division(uint256 number1,uint256 number2) public pure  returns (uint256){
        return number1/number2;
    }
}
